#include<stdio.h>
#include<string.h>

int nxt[100005];
char s[100005],t[100005];

void kmp_next(char *T,int *nt)
{
    nt[0]=-1;
    for(int i=0,j=-1,m=strlen(T);i<m;)
    {
        if(j==-1||T[i]==T[j])
        {
        	i++,j++;
			if(T[i]!=T[j])nt[i]=j;
			else nt[i]=nt[j];
        }
        else j=nt[j];
    }
}

int kmp(char *S,char *T,int *nt)
{
    kmp_next(T,nt);
    int ans=0,sn=strlen(S),tn=strlen(T);
    for(int i=0,j=0;i<sn;)
    {
        if(j==-1||S[i]==T[j])i++,j++;
        else j=nt[j];
        if(j==tn)ans++;
    }
    return ans;
}

int main()
{
	scanf("%s",s);
	int m;
	scanf("%d",&m);
	for(int i=1;i<=m;i++)
	{
		scanf("%s",t);
		printf("%s\n",kmp(s,t,nxt)>0?"yes":"no");
	}
	return 0;
}
