#include<stdio.h>
#include<iostream>
#include<algorithm>
using namespace std;
#include<string.h>
int match[1005], vis[1005];
int n, m;
int mp[1005][1005];
bool dfs(int u)
{
    for(int v=1;v<=m;v++)
    if(mp[u][v]&&!vis[v])
    {
        vis[v]=true;
        if( match[v] ==-1 || dfs(match[v]))
        {
            match[v]=u;
            return true;
        }
    }
    return false;
}
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
        scanf("%d%d", &n, &m);
        if(n>m) return 0*printf("NO\n");
        memset(mp, 0, sizeof(mp));
        for(int i=1;i<=n;i++)
        {
            int k,x;
            scanf("%d", &k);
            while(k--)
            {
                scanf("%d", &x);
                mp[i][x] = 1;
            }
        }
        int sum=0;
        memset(match,-1,sizeof(match));
        for (int i=1; i<=n; i++)
        {
            memset(vis,0,sizeof(vis));
            if (dfs(i)) sum++;
        }
        if(sum==n)printf("YES\n");
        else printf("NO\n");
    }
    return 0;
}
