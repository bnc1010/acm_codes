#include <stdio.h>
#include <algorithm>
using namespace std;
#define N 1000
int father[1000];
int init(int x){
    for(int i = 1;i <= x;i++)
    father[i] = i;
} 
int find(int v){
    return father[v] == v ? v : father[v] = find(father[v]);
}
void merge(int x,int y){
    int fx = find(x);
    int fy = find(y);
    if(fx != fy)
        father[fy] = fx;
}
int main()
{
    int n;
    int m;
    
    while(scanf("%d", &n), n)
    {
        scanf("%d", &m);
        init(n);
        int u,v;
        for(int i = 1;i<=m;i++)
        {
            scanf("%d%d", &u,&v);
            merge(u,v);
        }
        int ans = 0;
        for(int i = 1;i<=n;i++)
            if(father[i]==i)
                ans ++;
        printf("%d\n",ans - 1);        
    }
}
